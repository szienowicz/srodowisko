zad1 - wybór za pomocą 'case'

```c

#include<stdio.h>
main()
{
int wyb;
printf("podaj liczbę: \n1 wyswietla 'a'\n2 wyswietla 'c' \n3 wyswietla 'f' \n4 wyswietla 'koniec'\n");
scanf("%d",&wyb);
switch(wyb)
{

case 1:
printf("a\n");
break;

case 2:
printf("c\n");
break;

case 3:
printf("f\n");
break;

case 4:
printf("koniec\n");
break;
}
return 0;
}

```
